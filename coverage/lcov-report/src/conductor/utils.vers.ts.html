/**
 * オブジェクト配列を数値配列に変換
 * @param obj 変換対象の配列
 * @returns 変換された文字列または数値配列
 */
export function easyX(obj: any[] | undefined) {
  if (obj === undefined) return '';
  return obj.map(m => m === undefined ? '' : Math.floor(m))
}
